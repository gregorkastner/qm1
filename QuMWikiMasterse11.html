Mean, Median and Mode

Mean and median can be easily calculated by using these commands, resp. The mode comand exists, but gives back the basic structure of an object.

heights.even <- as.integer(round(heights/2)*2)

mean(heights.even)
## [1] 174.9995

median(heights.even)
## [1] 176

mode(heights.even)
## [1] "numeric"

How is it possible to calculate the mode?

First our data is round to even numbers to get more observations of a particular height. Then the table command gives back the number of observations of a particular height. The minus sign multiplies this numbers by (−1). Consequently the most observed height is the least observed height now. After sorting,there is no interest in the number of observations any more. Taking the first entry of this vector gives back the mode.

To be able to use this new tool more efficiently a myBoxViolinPlot function is created.

myBoxViolinPlot <-
  function(data, hoffset=0, outlier.method="quantile", alpha=0.001, coef=1.5) {
  if((outlier.method=="quantile") || (outlier.method=="both")) {
    box.marks <- quantile(
      data,
      probs=c(0, (alpha/2), .25, .5, .75, (1-(alpha/2)), 1)
    )
    myBoxplot.Out <- data[(data < box.marks[2]) | (data > box.marks[6])]
    plot(NA, ylim=c(box.marks["0%"], box.marks["100%"]),
         xaxt="n", xlab=NA,
         ylab="Boxplot")
    lines(c(.9, 1.0)+hoffset, c(box.marks[2],box.marks[2]))
    lines(c(.8, 1.0)+hoffset, c(box.marks["25%"],box.marks["25%"]))
    lines(c(.8, 1.0)+hoffset, c(box.marks["50%"],box.marks["50%"]), lty=3)
    lines(c(.8, 1.0)+hoffset, c(box.marks["75%"],box.marks["75%"]))
    lines(c(.9, 1.0)+hoffset, c(box.marks[6],box.marks[6]))
    lines(c(1, 1)+hoffset, c(box.marks[2],box.marks["25%"]), lty=2)
    lines(c(1, 1)+hoffset, c(box.marks["75%"],box.marks[6]), lty=2)
    lines(c(.8, .8)+hoffset, c(box.marks["25%"],box.marks["75%"]))
    lines(c(1.0, 1.0)+hoffset, c(box.marks["25%"],box.marks["75%"]))
    points(rep(1, length(myBoxplot.Out))+hoffset, myBoxplot.Out)
    data.density <- density(data)
    lines(data.density$y+1+hoffset, data.density$x)
  }
  if((outlier.method=="relToBox") || (outlier.method=="both")) {
    myBoxplot.stats <- boxplot.stats(data, coef=coef)
    plot(NA, ylim=c(min(myBoxplot.stats$stats[1], myBoxplot.stats$out),
                    max(myBoxplot.stats$stats[5], myBoxplot.stats$out)),
         xaxt="n", xlab=NA,
         ylab="Boxplot")
    lines(c(.9, 1.0)+hoffset, c(myBoxplot.stats$stats[1], myBoxplot.stats$stats[1]))
    lines(c(.8, 1.0)+hoffset, c(myBoxplot.stats$stats[2], myBoxplot.stats$stats[2]))
    lines(c(.8, 1.0)+hoffset, c(myBoxplot.stats$stats[3], myBoxplot.stats$stats[3]), lty=3)
    lines(c(.8, 1.0)+hoffset, c(myBoxplot.stats$stats[4], myBoxplot.stats$stats[4]))
    lines(c(.9, 1.0)+hoffset, c(myBoxplot.stats$stats[5], myBoxplot.stats$stats[5]))
    lines(c(1, 1)+hoffset, c(myBoxplot.stats$stats[1], myBoxplot.stats$stats[2]), lty=2)
    lines(c(1, 1)+hoffset, c(myBoxplot.stats$stats[4], myBoxplot.stats$stats[5]), lty=2)
    lines(c(.8, .8)+hoffset, c(myBoxplot.stats$stats[2], myBoxplot.stats$stats[4]))
    lines(c(1.0, 1.0)+hoffset, c(myBoxplot.stats$stats[2], myBoxplot.stats$stats[4]))
    points(rep(1, length(myBoxplot.stats$out))+hoffset, myBoxplot.stats$out)
    data.density <- density(data)
    lines(data.density$y+1+hoffset, data.density$x)
  } else {
    stop("Please specifiy a valid outlier method (either quantile, relToBox or both)")
  }
}

myBoxViolinPlot(heights, outlier.method="both")
